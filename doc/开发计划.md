## 一、项目概述

### 1.1 项目名称
**TW AI 节能器** - 本地可部署的 AI 代理中枢

### 1.2 项目目标
为个人开发者提供轻量级、本地部署的 AI 代理服务，实现：
- 多模型统一调度（OpenAI/Ollama/硅基流动/LM Studio）
- 成本优化（小模型优先，失败自动切换大模型）
- 会话沉淀与 Skill 自动化生成
- 跨平台支持（Windows/Linux/macOS）

### 1.3 技术栈

| 层级 | 技术 |
|------|------|
| 后端框架 | Python FastAPI |
| 数据库 | MongoDB 7 |
| 向量库 | Qdrant |
| 前端框架 | Vue 3 + Element Plus |
| 部署 | Docker Compose |
| 操作系统 | Windows / Linux / macOS |

**2.2.1 代理转发服务**

```python
# proxy-api/app/services/proxy_service.py
from typing import Optional
import httpx
from app.core.config import settings

class ProxyService:
    def __init__(self):
        self.fail_count = {}  # {proxy_key: count}
    
    async def forward_request(
        self,
        proxy_key: str,
        model: str,
        messages: list,
        stream: bool = False
    ) -> dict:
        # 1. 获取连接配置
        connection = await self.get_connection(proxy_key)
        
        # 2. 尝试小模型
        small_model = connection.small_model
        response = await self.call_model(
            small_model.base_url,
            small_model.api_key,
            small_model.name,
            messages,
            stream
        )
        
        # 3. 检查超时
        if response.get("timeout"):
            self.fail_count[proxy_key] = self.fail_count.get(proxy_key, 0) + 1
            
            # 4. 检查阈值
            if self.fail_count[proxy_key] >= settings.strategy.fail_threshold:
                # 切换大模型
                self.fail_count[proxy_key] = 0
                big_model = connection.big_model
                response = await self.call_model(
                    big_model.base_url,
                    big_model.api_key,
                    big_model.name,
                    messages,
                    stream
                )
        
        return response
    
    async def call_model(
        self,
        base_url: str,
        api_key: str,
        model: str,
        messages: list,
        stream: bool = False
    ) -> dict:
        async with httpx.AsyncClient(timeout=settings.app.model_timeout/1000) as client:
            response = await client.post(
                f"{base_url}/chat/completions",
                headers={"Authorization": f"Bearer {api_key}"},
                json={
                    "model": model,
                    "messages": messages,
                    "stream": stream
                }
            )
            return response.json()
```

```python
# proxy-api/app/infrastructure/repositories/session_repository_impl.py
from app.domain.repositories.session_repository import SessionRepository
from app.infrastructure.database.mongodb import mongodb
from bson import ObjectId

class SessionRepositoryImpl(SessionRepository):
    collection = mongodb.db["sessions"]
    
    async def get_all(self, page: int = 1, limit: int = 20) -> tuple[list, int]:
        skip = (page - 1) * limit
        cursor = self.collection.find().skip(skip).limit(limit)
        total = await self.collection.count_documents({})
        return await cursor.to_list(length=limit), total
    
    async def get_by_id(self, session_id: str) -> dict:
        return await self.collection.find_one({"session_id": session_id})
    
    async def create(self, session: dict) -> str:
        result = await self.collection.insert_one(session)
        return str(result.inserted_id)
    
    async def update(self, session_id: str, session: dict) -> None:
        await self.collection.update_one(
            {"session_id": session_id},
            {"$set": session}
        )
    
    async def delete(self, session_id: str) -> None:
        await self.collection.delete_one({"session_id": session_id})
    
    async def batch_delete(self, session_ids: list[str]) -> int:
        result = await self.collection.delete_many({
            "session_id": {"$in": session_ids}
        })
        return result.deleted_count
```

```python
# proxy-api/app/services/skill_service.py
class SkillService:
    async def create_version(self, base_id: str, data: dict) -> int:
        metadata = await skill_repository.get_metadata(base_id)
        
        # 获取当前最大版本号
        max_version = max([v["version_id"] for v in metadata["versions"]])
        new_version = max_version + 1
        
        # 创建新版本
        await skill_repository.create_version(base_id, {
            "version_id": new_version,
            "status": "draft",
            "created_at": datetime.utcnow().isoformat(),
            "created_by": "system" or "user",
            "source_session_id": data.get("session_id"),
            "change_reason": data.get("change_reason", "optimization")
        })
        
        return new_version
    
    async def publish(self, base_id: str, version_id: int) -> None:
        # 1. 更新版本状态为 published
        await skill_repository.update_version_status(
            base_id, version_id, "published"
        )
        
        # 2. 更新 active_version_id
        await skill_repository.set_active_version(base_id, version_id)
        
        # 3. 将旧版本标记为 deprecated
        for v in metadata["versions"]:
            if v["version_id"] != version_id:
                await skill_repository.update_version_status(
                    base_id, v["version_id"], "deprecated"
                )
        
        # 4. 向量化
        await vector_service.vectorize_skill(base_id, version_id)
    
    async def rollback(self, base_id: str, version_id: int) -> None:
        # 回滚到指定版本
        await self.publish(base_id, version_id)
```

**5.2.1 主题切换实现**

```javascript
// web-ui/src/stores/theme.js
import { defineStore } from 'pinia'
import axios from 'axios'

export const useThemeStore = defineStore('theme', {
  state: () => ({
    isDark: false,
    colors: {
      light: {
        primary: '#409EFF',
        bg: '#F5F7FA',
        card: '#FFFFFF',
        text: '#303133'
      },
      dark: {
        primary: '#409EFF',
        bg: '#141414',
        card: '#1D1D1D',
        text: '#E5EAF3'
      }
    }
  }),
  
  actions: {
    async loadTheme() {
      try {
        const response = await axios.get('/api/v1/config')
        this.isDark = response.data.theme === 'dark'
        this.applyTheme()
      } catch (error) {
        console.error('加载主题失败:', error)
      }
    },
    
    toggleTheme() {
      this.isDark = !this.isDark
      this.applyTheme()
    },
    
    applyTheme() {
      const colors = this.isDark ? this.colors.dark : this.colors.light
      
      // 应用 CSS 变量
      Object.entries(colors).forEach(([key, value]) => {
        document.documentElement.style.setProperty(`--color-${key}`, value)
      })
      
      // 切换 body 类
      if (this.isDark) {
        document.body.classList.add('dark')
      } else {
        document.body.classList.remove('dark')
      }
    }
  }
})
```

**6.2.2 baseskill 管理 API**

```python
# proxy-api/app/api/admin/baseskills.py
import os
import shutil
import zipfile
from fastapi import APIRouter, HTTPException, UploadFile
from pathlib import Path

router = APIRouter(prefix="/api/v1/baseskills", tags=["baseskill 管理"])

BASESKILL_PATH = Path("./data/baseskill")
SYSTEM_BASESKILL_PATH = Path("./baseskill")

@router.get("/")
async def list_baseskills():
    # 列出系统内置和用户自定义
    result = []
    
    # 系统内置
    for name in SYSTEM_BASESKILL_PATH.iterdir():
        if name.is_dir():
            result.append({
                "id": f"system-{name.name}",
                "name": name.name,
                "type": "system",
                "path": str(name)
            })
    
    # 用户自定义
    if BASESKILL_PATH.exists():
        for name in BASESKILL_PATH.iterdir():
            if name.is_dir():
                result.append({
                    "id": f"custom-{name.name}",
                    "name": name.name,
                    "type": "custom",
                    "path": str(name)
                })
    
    return result

@router.post("/")
async def create_baseskill(data: dict):
    name = data["name"]
    path = BASESKILL_PATH / name
    path.mkdir(parents=True, exist_ok=True)
    
    # 创建文件
    (path / "SKILL.md").write_text(data["skill_md"])
    (path / "prompt-template.md").write_text(data["prompt_template"])
    (path / "rules.yaml").write_text(data["rules_yaml"])
    
    return {"id": f"custom-{name}", "name": name}

@router.post("/upload")
async def upload_baseskill(file: UploadFile):
    import tempfile
    
    with tempfile.TemporaryDirectory() as tmp_dir:
        tmp_path = Path(tmp_dir) / file.filename
        tmp_path.write_bytes(await file.read())
        
        # 解压
        extract_path = BASESKILL_PATH / file.filename.replace(".zip", "")
        with zipfile.ZipFile(tmp_path, 'r') as zip_ref:
            zip_ref.extractall(extract_path)
        
        return {"name": extract_path.name}

@router.delete("/{id}")
async def delete_baseskill(id: str):
    if id.startswith("system-"):
        raise HTTPException(status_code=403, detail="Cannot delete system baseskill")
    
    name = id.replace("custom-", "")
    path = BASESKILL_PATH / name
    if path.exists():
        shutil.rmtree(path)

@router.post("/reload")
async def reload_baseskills():
    # 清空内存中的缓存，重新加载
    return {"status": "reloaded"}
```
